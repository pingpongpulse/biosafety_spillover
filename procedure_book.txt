Chapter 0: Project Overview
Read this chapter before doing anything else. It explains the full picture of what you are building and why.
What is BioRiskNet?
BioRiskNet is a dual-objective machine learning framework that does two things simultaneously:
Model 1 — WHO Risk Group Classifier: Given a pathogen's biological metadata (host type, transmission mode, genome type, etc.), predict whether it belongs to Risk Group 1, 2, 3, or 4.
Model 2 — Zoonotic Spillover Scorer: Given a virus's known host associations, predict the probability that it will spill over from animals into humans.
SHAP Explainability: For every prediction, show which biological features drove the decision and by how much.
Why Is This Novel?
No published paper has built a multi-class (RG1-4) WHO Risk Group classifier from metadata alone with SHAP explanations. Existing tools are either binary or genome-sequence based. That gap is your contribution.

Step 1.4 — Scrape ABSA Risk Group Database
This is the most important dataset for Model 1. ABSA lists pathogens with WHO Risk Group labels. There is no bulk download, so we scrape it.
WARNING: Run the scraper slowly (with time.sleep). If you scrape too fast, the server may block you. The sleep(2) in the script handles this.
Script: scripts/scrape_absa.py
import requests
from bs4 import BeautifulSoup
import pandas as pd
import time
BASE_URL = 'https://absa.org/biosafety-resources/risk-group-database/'
records = []
def scrape_page(url):
    r = requests.get(url, timeout=15)
    if r.status_code != 200:
        print(f'Failed: {url}')
        return None
    return BeautifulSoup(r.text, 'html.parser')
def extract_records(soup):
    rows = soup.find_all('tr')
    for row in rows:
        cols = row.find_all('td')
        if len(cols) >= 3:
            records.append({
                'organism': cols[0].get_text(strip=True),
                'risk_group': cols[1].get_text(strip=True),
                'notes': cols[2].get_text(strip=True),
            })
# Scrape main page
soup = scrape_page(BASE_URL)
if soup:
    extract_records(soup)
    # Find pagination links
    pages = soup.find_all('a', href=True)
    page_links = [a['href'] for a in pages if 'page' in a['href'].lower()]
    for link in page_links:
        full_url = link if link.startswith('http') else BASE_URL + link
        print(f'Scraping: {full_url}')
        soup2 = scrape_page(full_url)
        if soup2:
            extract_records(soup2)
        time.sleep(2)  # Be polite — wait 2 seconds between pages
df = pd.DataFrame(records).drop_duplicates()
df.to_csv('data/raw/absa_raw.csv', index=False)
print(f'Done. Saved {len(df)} records to absa_raw.csv')

import pandas as pd
import numpy as np
# ── Load datasets ──────────────────────────────────────────
absa = pd.read_csv('data/raw/absa_raw.csv')
epatho = pd.read_csv('data/raw/epathogen_clean.csv')
bmbl = pd.read_csv('data/raw/bmbl_extracted.csv')
olival_v = pd.read_csv('data/raw/olival_viruses.csv')
olival_a = pd.read_csv('data/raw/olival_associations.csv')
# ── Standardize column names ───────────────────────────────
absa = absa.rename(columns={'organism': 'name', 'risk_group': 'rg_label'})
epatho = epatho.rename(columns={'Organism': 'name', 'Risk Group': 'rg_label'})
bmbl = bmbl.rename(columns={'organism': 'name', 'bsl_level': 'rg_label'})
# ── Normalize Risk Group labels to integers 1-4 ────────────
def normalize_rg(val):
    val = str(val).strip().upper()
    mapping = {
        'RG1': 1, 'RG2': 2, 'RG3': 3, 'RG4': 4,
        'RISK GROUP 1': 1, 'RISK GROUP 2': 2,
        'RISK GROUP 3': 3, 'RISK GROUP 4': 4,
        '1': 1, '2': 2, '3': 3, '4': 4,
        'BSL-1': 1, 'BSL-2': 2, 'BSL-3': 3, 'BSL-4': 4,
        'BSL 1': 1, 'BSL 2': 2, 'BSL 3': 3, 'BSL 4': 4,
    }
    return mapping.get(val, np.nan)
for df in [absa, epatho, bmbl]:
    df['rg_label'] = df['rg_label'].apply(normalize_rg)
# ── Combine classification datasets ───────────────────────
combined_rg = pd.concat([
    absa[['name', 'rg_label']],
    epatho[['name', 'rg_label']],
    bmbl[['name', 'rg_label']],
], ignore_index=True)
# Drop rows with missing labels
combined_rg = combined_rg.dropna(subset=['rg_label'])
combined_rg['rg_label'] = combined_rg['rg_label'].astype(int)
# Handle duplicates: where sources disagree, take the max (more conservative)
combined_rg = combined_rg.groupby('name', as_index=False)['rg_label'].max()
print('Risk Group dataset shape:', combined_rg.shape)
print('Label distribution:')
print(combined_rg['rg_label'].value_counts())
# ── Add biological features ────────────────────────────────
# These features need to be manually coded or looked up from literature
# For now, create a template that Person 2 can fill with encoding
feature_cols = ['genome_type', 'transmission_route', 'host_range',
                'environmental_stability', 'treatment_available',
                'infectious_dose', 'zoonotic']
for col in feature_cols:
    combined_rg[col] = np.nan  # Person 2 will encode these
# ── Save final datasets ────────────────────────────────────
combined_rg.to_csv('data/processed/rg_dataset.csv', index=False)
olival_v.to_csv('data/processed/spillover_dataset.csv', index=False)
print('\nSaved rg_dataset.csv and spillover_dataset.csv to data/processed/')
print('DONE — hand off to Person 2 now')

Step 2.2 — Encode Features
Machine learning models need numbers, not text. This script converts all text features into numeric codes.
Script: scripts/encode_features.py
import pandas as pd
import numpy as np
from sklearn.preprocessing import LabelEncoder
df = pd.read_csv('data/processed/rg_dataset_features.csv')
print('Loaded dataset shape:', df.shape)
print('Columns:', df.columns.tolist())
# Drop rows where label is missing
df = df.dropna(subset=['rg_label'])
# Define categorical feature columns
cat_cols = ['genome_type', 'transmission_route', 'host_range',
            'environmental_stability', 'treatment_available',
            'infectious_dose', 'zoonotic']
# Encode each categorical column
encoders = {}
for col in cat_cols:
    if col in df.columns:
        df[col] = df[col].fillna('unknown')
        le = LabelEncoder()
        df[col + '_enc'] = le.fit_transform(df[col])
        encoders[col] = le
        print(f'Encoded {col}: {dict(zip(le.classes_, le.transform(le.classes_)))}')
# Select feature columns for model
feature_cols = [c + '_enc' for c in cat_cols if c in df.columns]
X = df[feature_cols]
y = df['rg_label'].astype(int) - 1  # Convert 1-4 to 0-3 for XGBoost
print('\nFeature matrix shape:', X.shape)
print('Label distribution (0=RG1, 1=RG2, 2=RG3, 3=RG4):')
print(y.value_counts().sort_index())
# Save encoded dataset
X['rg_label'] = y
X.to_csv('data/processed/encoded_dataset.csv', index=False)
print('\nSaved encoded_dataset.csv')

Step 2.3 — Handle Class Imbalance with SMOTE
RG3 and RG4 organisms are rare. If you train without fixing this, the model will just predict RG2 for everything. SMOTE creates synthetic examples of the minority classes.
Script: scripts/apply_smote.py
import pandas as pd
from imblearn.over_sampling import SMOTE
from collections import Counter
df = pd.read_csv('data/processed/encoded_dataset.csv')
X = df.drop('rg_label', axis=1)
y = df['rg_label']
print('Before SMOTE:', Counter(y))
smote = SMOTE(random_state=42, k_neighbors=3)
X_resampled, y_resampled = smote.fit_resample(X, y)
print('After SMOTE:', Counter(y_resampled))
# Save resampled dataset
df_resampled = pd.DataFrame(X_resampled, columns=X.columns)
df_resampled['rg_label'] = y_resampled
df_resampled.to_csv('data/processed/smote_dataset.csv', index=False)

import pandas as pd
import numpy as np
import xgboost as xgb
from sklearn.model_selection import StratifiedKFold, cross_val_score
from sklearn.metrics import (classification_report, confusion_matrix,
                             accuracy_score, f1_score)
import pickle
import json
# Load data
df = pd.read_csv('data/processed/smote_dataset.csv')
X = df.drop('rg_label', axis=1)
y = df['rg_label']
# ── Define model ──────────────────────────────────────────
model = xgb.XGBClassifier(
    n_estimators=300,
    max_depth=6,
    learning_rate=0.05,
    subsample=0.8,
    colsample_bytree=0.8,
    use_label_encoder=False,
    eval_metric='mlogloss',
    random_state=42,
    num_class=4
)
# ── 5-fold stratified cross validation ────────────────────
cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
cv_scores = cross_val_score(model, X, y, cv=cv, scoring='f1_macro')
print('Cross-validation F1-macro scores:', cv_scores)
print(f'Mean: {cv_scores.mean():.3f} +/- {cv_scores.std():.3f}')
# ── Train final model on all data ─────────────────────────
model.fit(X, y)
# ── Evaluate on training set (for paper reporting) ────────
y_pred = model.predict(X)
print('\nClassification Report:')
print(classification_report(y, y_pred,
      target_names=['RG1','RG2','RG3','RG4']))
# ── Save model and results ────────────────────────────────
with open('results/xgboost_model.pkl', 'wb') as f:
    pickle.dump(model, f)
results = {
    'cv_f1_scores': cv_scores.tolist(),
    'cv_mean': float(cv_scores.mean()),
    'cv_std': float(cv_scores.std()),
    'feature_names': X.columns.tolist()
}
with open('results/model_results.json', 'w') as f:
    json.dump(results, f, indent=2)
print('\nModel saved to results/xgboost_model.pkl')
print('Results saved to results/model_results.json')

Step 3.3 — Build the Spillover Risk Model

Script: scripts/train_spillover.py
import pandas as pd
import numpy as np
import xgboost as xgb
from sklearn.model_selection import cross_val_score, StratifiedKFold
from sklearn.metrics import roc_auc_score, classification_report
from sklearn.preprocessing import LabelEncoder
from imblearn.over_sampling import SMOTE
import pickle
# Load Olival data
viruses = pd.read_csv('data/raw/olival_viruses.csv')
assoc = pd.read_csv('data/raw/olival_associations.csv')
print('Viruses columns:', viruses.columns.tolist())
print('Associations columns:', assoc.columns.tolist())
# Create binary spillover label
# 'IsZoonotic' or similar column — check actual column name
# 1 = known to infect humans, 0 = not known
if 'IsZoonotic' in viruses.columns:
    viruses['spillover'] = (viruses['IsZoonotic'] == 1).astype(int)
elif 'hHostCount' in viruses.columns:
    viruses['spillover'] = (viruses['hHostCount'] > 0).astype(int)
else:
    # Fallback: check associations for human host
    human_viruses = assoc[assoc['hHostName'].str.contains('Homo sapiens', na=False)]['vVirusNameCorrected'].unique()
    viruses['spillover'] = viruses['vVirusNameCorrected'].isin(human_viruses).astype(int)
print('Spillover distribution:', viruses['spillover'].value_counts().to_dict())
# Select features
# These are numeric columns in Olival dataset
feature_candidates = [
    'vDomains', 'vNonSegmented', 'vEnveloped',
    'hHostCount', 'hWildDomestic', 'hSpeciesAverageMass',
    'phyloDistCG', 'NaiveBAYES'
]
available = [c for c in feature_candidates if c in viruses.columns]
print('Available features:', available)
df = viruses[available + ['spillover']].dropna()
X = df[available]
y = df['spillover']
print('Dataset size after cleaning:', len(df))
# Handle class imbalance
smote = SMOTE(random_state=42, k_neighbors=3)
X_res, y_res = smote.fit_resample(X, y)
# Train spillover model
model = xgb.XGBClassifier(
    n_estimators=200,
    max_depth=5,
    learning_rate=0.05,
    use_label_encoder=False,
    eval_metric='logloss',
    random_state=42
)
cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
auc_scores = cross_val_score(model, X_res, y_res, cv=cv, scoring='roc_auc')
print(f'Spillover AUC: {auc_scores.mean():.3f} +/- {auc_scores.std():.3f}')
model.fit(X_res, y_res)
with open('results/spillover_model.pkl', 'wb') as f:
    pickle.dump(model, f)
print('Spillover model saved to results/spillover_model.pkl')
